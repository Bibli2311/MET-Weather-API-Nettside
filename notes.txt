async function fetchData(url)
{
  fetch(url)
  .then(async (resp) => {
    if (resp.ok) {
        
        let respTxt = await resp.text()
        return new window.DOMParser().parseFromString(respTxt, "text/xml");
    }
    throw new Error("Error with network");
  })
  .catch(error => console.error(error))
}
